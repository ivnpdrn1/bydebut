Perfecto. Vamos a construir **Security & Collaboration Engine v1.0** en dos pasos:

1. **Tablas de colaboración y permisos**
2. **RLS / Security Policies**

Primero ejecuta este bloque en Supabase SQL Editor.

```sql
-- =====================================================
-- byDebut Security & Collaboration Engine v1.0
-- Step 1: Collaboration + Permission Tables
-- =====================================================

-- 1. Book visibility
do $$ begin
  create type book_visibility as enum (
    'private',
    'unlisted',
    'public',
    'community',
    'licensed'
  );
exception when duplicate_object then null;
end $$;

-- 2. Collaborator roles
do $$ begin
  create type collaborator_role as enum (
    'viewer',
    'commenter',
    'contributor',
    'reviewer',
    'editor',
    'publisher',
    'owner'
  );
exception when duplicate_object then null;
end $$;

-- 3. Lineage visibility
do $$ begin
  create type lineage_visibility as enum (
    'private',
    'unlisted',
    'public'
  );
exception when duplicate_object then null;
end $$;

-- 4. Add visibility to books
alter table public.books
add column if not exists visibility book_visibility not null default 'private';

-- 5. Add QR / Story Passport fields to books
alter table public.books
add column if not exists slug text;

alter table public.books
add column if not exists qr_entry_enabled boolean not null default false;

alter table public.books
add column if not exists passport_summary text;

alter table public.books
add column if not exists copyright_owner text;

alter table public.books
add column if not exists rights_status text default 'owner_uploaded';

alter table public.books
add column if not exists allow_public_lineages boolean not null default false;

alter table public.books
add column if not exists allow_reader_versions boolean not null default true;

alter table public.books
add column if not exists canonical_version_id uuid references public.story_versions(id) on delete set null;

-- 6. Add lineage fields to story_versions
alter table public.story_versions
add column if not exists lineage_visibility lineage_visibility not null default 'private';

alter table public.story_versions
add column if not exists lineage_name text;

alter table public.story_versions
add column if not exists lineage_slug text;

alter table public.story_versions
add column if not exists is_canonical boolean not null default false;

alter table public.story_versions
add column if not exists is_published boolean not null default false;

alter table public.story_versions
add column if not exists published_at timestamptz;

-- 7. Book collaborators
create table if not exists public.book_collaborators (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  user_id uuid not null references public.profiles(id) on delete cascade,

  role collaborator_role not null default 'viewer',

  invited_by uuid references public.profiles(id) on delete set null,
  accepted_at timestamptz,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  unique(book_id, user_id)
);

-- 8. Version collaborators
create table if not exists public.version_collaborators (
  id uuid primary key default gen_random_uuid(),

  version_id uuid not null references public.story_versions(id) on delete cascade,
  user_id uuid not null references public.profiles(id) on delete cascade,

  role collaborator_role not null default 'viewer',

  invited_by uuid references public.profiles(id) on delete set null,
  accepted_at timestamptz,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  unique(version_id, user_id)
);

-- 9. Comments
create table if not exists public.comments (
  id uuid primary key default gen_random_uuid(),

  book_id uuid references public.books(id) on delete cascade,
  version_id uuid references public.story_versions(id) on delete cascade,
  chapter_id uuid references public.chapters(id) on delete cascade,

  user_id uuid not null references public.profiles(id) on delete cascade,

  content text not null,
  is_private boolean not null default false,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

-- 10. Bookmarks / Favorites
create table if not exists public.bookmarks (
  id uuid primary key default gen_random_uuid(),

  user_id uuid not null references public.profiles(id) on delete cascade,
  book_id uuid references public.books(id) on delete cascade,
  version_id uuid references public.story_versions(id) on delete cascade,

  created_at timestamptz not null default now(),

  unique(user_id, book_id, version_id)
);

-- 11. Reading progress
create table if not exists public.reading_progress (
  id uuid primary key default gen_random_uuid(),

  user_id uuid not null references public.profiles(id) on delete cascade,
  book_id uuid not null references public.books(id) on delete cascade,
  version_id uuid references public.story_versions(id) on delete cascade,
  chapter_id uuid references public.chapters(id) on delete set null,

  progress_percent numeric not null default 0,
  last_position jsonb default '{}'::jsonb,

  updated_at timestamptz not null default now(),

  unique(user_id, book_id, version_id)
);

-- 12. Story Passport analytics
create table if not exists public.story_passport_visits (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  user_id uuid references public.profiles(id) on delete set null,

  source text default 'qr',
  metadata jsonb default '{}'::jsonb,

  created_at timestamptz not null default now()
);

-- 13. Lineage activity
create table if not exists public.lineage_activity (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  version_id uuid references public.story_versions(id) on delete cascade,
  user_id uuid references public.profiles(id) on delete set null,

  activity_type text not null,
  metadata jsonb default '{}'::jsonb,

  created_at timestamptz not null default now()
);

-- 14. Helpful indexes
create index if not exists idx_books_slug
on public.books(slug);

create index if not exists idx_books_visibility
on public.books(visibility);

create index if not exists idx_book_collaborators_book_id
on public.book_collaborators(book_id);

create index if not exists idx_book_collaborators_user_id
on public.book_collaborators(user_id);

create index if not exists idx_version_collaborators_version_id
on public.version_collaborators(version_id);

create index if not exists idx_version_collaborators_user_id
on public.version_collaborators(user_id);

create index if not exists idx_comments_book_id
on public.comments(book_id);

create index if not exists idx_comments_version_id
on public.comments(version_id);

create index if not exists idx_bookmarks_user_id
on public.bookmarks(user_id);

create index if not exists idx_story_passport_visits_book_id
on public.story_passport_visits(book_id);

create index if not exists idx_lineage_activity_book_id
on public.lineage_activity(book_id);

-- 15. updated_at triggers for new tables
drop trigger if exists set_book_collaborators_updated_at on public.book_collaborators;
create trigger set_book_collaborators_updated_at
before update on public.book_collaborators
for each row execute function public.set_updated_at();

drop trigger if exists set_version_collaborators_updated_at on public.version_collaborators;
create trigger set_version_collaborators_updated_at
before update on public.version_collaborators
for each row execute function public.set_updated_at();

drop trigger if exists set_comments_updated_at on public.comments;
create trigger set_comments_updated_at
before update on public.comments
for each row execute function public.set_updated_at();
```

Cuando te dé **Success. No rows returned**, seguimos con el bloque de **RLS / Security Policies**.
