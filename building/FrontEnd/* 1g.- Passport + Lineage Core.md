    'draft',
    'active',
    'featured',
    'canonical_candidate',
    'canonical',
    'archived'
  );
exception when duplicate_object then null;
end $$;

create table if not exists public.story_passports (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null unique references public.books(id) on delete cascade,

  passport_slug text unique not null,
  qr_token text unique not null default encode(gen_random_bytes(12), 'hex'),

  public_title text not null,
  public_subtitle text,
  public_description text,

  original_author text,
  copyright_owner text,
  publisher_name text,

  rights_statement text,
  license_terms text,
  allowed_versioning boolean not null default true,

  total_lineages integer not null default 0,
  total_versions integer not null default 0,
  total_visits integer not null default 0,
  total_forks integer not null default 0,

  is_public boolean not null default false,
  is_featured boolean not null default false,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.story_lineages (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  root_version_id uuid not null references public.story_versions(id) on delete cascade,
  created_by uuid not null references public.profiles(id) on delete cascade,

  name text not null,
  slug text not null,
  description text,

  lineage_type text default 'reader_version',
  status lineage_status not null default 'draft',

  visibility lineage_visibility not null default 'private',

  fork_count integer not null default 0,
  view_count integer not null default 0,
  like_count integer not null default 0,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  unique(book_id, slug)
);

alter table public.story_versions
add column if not exists lineage_id uuid references public.story_lineages(id) on delete set null;

alter table public.story_versions
add column if not exists forked_from_version_id uuid references public.story_versions(id) on delete set null;

alter table public.story_versions
add column if not exists ai_model_used text;

alter table public.story_versions
add column if not exists reviewer_score numeric default 0;

alter table public.story_versions
add column if not exists public_read_count integer not null default 0;

alter table public.story_versions
add column if not exists public_like_count integer not null default 0;

create table if not exists public.lineage_forks (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  source_lineage_id uuid references public.story_lineages(id) on delete set null,
  target_lineage_id uuid references public.story_lineages(id) on delete cascade,

  source_version_id uuid references public.story_versions(id) on delete set null,
  target_version_id uuid references public.story_versions(id) on delete cascade,

  forked_by uuid not null references public.profiles(id) on delete cascade,

  fork_reason text,
  fork_prompt text,

  created_at timestamptz not null default now()
);

create table if not exists public.version_likes (
  id uuid primary key default gen_random_uuid(),

  user_id uuid not null references public.profiles(id) on delete cascade,
  version_id uuid not null references public.story_versions(id) on delete cascade,

  created_at timestamptz not null default now(),

  unique(user_id, version_id)
);

create table if not exists public.lineage_likes (
  id uuid primary key default gen_random_uuid(),

  user_id uuid not null references public.profiles(id) on delete cascade,
  lineage_id uuid not null references public.story_lineages(id) on delete cascade,

  created_at timestamptz not null default now(),

  unique(user_id, lineage_id)
);

create index if not exists idx_story_passports_book_id

create index if not exists idx_story_passports_slug
on public.story_passports(passport_slug);

create index if not exists idx_story_passports_qr_token

create index if not exists idx_story_lineages_book_id
on public.story_lineages(book_id);

create index if not exists idx_story_lineages_slug

on public.story_versions(lineage_id);

create index if not exists idx_lineage_forks_book_id
on public.lineage_forks(book_id);

drop trigger if exists set_story_passports_updated_at on public.story_passports;
create trigger set_story_passports_updated_at
before update on public.story_passports
for each row execute function public.set_updated_at();

drop trigger if exists set_story_lineages_updated_at on public.story_lineages;
create trigger set_story_lineages_updated_at
before update on public.story_lineages
for each row execute function public.set_updated_at();
