Sí. Vamos con **Story Intelligence Module v1.0**.

Ejecuta este bloque en **Supabase → SQL Editor → New Query**.

```sql
-- =====================================================
-- byDebut Story Intelligence Module v1.0
-- Step 1: Intelligence Tables
-- =====================================================

-- 1. Generic updated_at trigger function
create or replace function public.set_updated_at()
returns trigger as $$
begin
  new.updated_at = now();
  return new;
end;
$$ language plpgsql;


-- 2. Story analysis summary
create table if not exists public.story_analysis (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  created_by uuid not null references public.profiles(id) on delete cascade,

  summary text,
  central_conflict text,
  narrative_style text,
  tone text,
  genre_detected text,
  language_detected text,

  themes jsonb default '[]'::jsonb,
  narrative_arc jsonb default '{}'::jsonb,
  ai_model text default 'gemini',

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  unique(book_id)
);


-- 3. Story chunks for RAG
create table if not exists public.story_chunks (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  chapter_id uuid references public.chapters(id) on delete cascade,
  version_id uuid references public.story_versions(id) on delete cascade,

  chunk_index integer not null,
  content text not null,
  token_count integer default 0,

  metadata jsonb default '{}'::jsonb,

  created_at timestamptz not null default now(),

  unique(book_id, chapter_id, version_id, chunk_index)
);


-- 4. Embeddings
-- MVP dimension: 768. If later Gemini embedding dimension changes, we can migrate.
create table if not exists public.story_embeddings (
  id uuid primary key default gen_random_uuid(),

  chunk_id uuid not null references public.story_chunks(id) on delete cascade,
  book_id uuid not null references public.books(id) on delete cascade,

  embedding vector(768) not null,
  model text default 'gemini-embedding',

  created_at timestamptz not null default now()
);


-- 5. Story entities: characters, locations, objects, concepts
create table if not exists public.story_entities (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,

  entity_type text not null check (
    entity_type in (
      'character',
      'location',
      'object',
      'organization',
      'concept',
      'theme'
    )
  ),

  name text not null,
  description text,
  importance_score numeric default 0,

  attributes jsonb default '{}'::jsonb,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  unique(book_id, entity_type, name)
);


-- 6. Relationships between entities
create table if not exists public.story_relationships (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,

  source_entity_id uuid not null references public.story_entities(id) on delete cascade,
  target_entity_id uuid not null references public.story_entities(id) on delete cascade,

  relationship_type text not null,
  description text,
  strength numeric default 0,

  created_at timestamptz not null default now(),

  unique(source_entity_id, target_entity_id, relationship_type)
);


-- 7. Story events
create table if not exists public.story_events (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  chapter_id uuid references public.chapters(id) on delete set null,

  event_order integer not null,
  title text not null,
  description text,
  emotional_impact text,
  importance_score numeric default 0,

  involved_entities jsonb default '[]'::jsonb,

  created_at timestamptz not null default now(),

  unique(book_id, event_order)
);


-- 8. Prompt context memory
create table if not exists public.prompt_contexts (
  id uuid primary key default gen_random_uuid(),

  book_id uuid not null references public.books(id) on delete cascade,
  version_id uuid references public.story_versions(id) on delete cascade,

  context_type text not null check (
    context_type in (
      'summary',
      'character_memory',
      'timeline',
      'style_guide',
      'continuity_rules',
      'generation_context'
    )
  ),

  content text not null,
  metadata jsonb default '{}'::jsonb,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);
```

Después de ejecutarlo, deberías ver **Success. No rows returned**.

El siguiente bloque será para crear **índices + búsqueda vectorial**, que es lo que permitirá a Gemini recuperar contexto relevante antes de generar versiones.
