```sql
create index if not exists idx_books_owner_id

create index if not exists idx_chapters_book_id
on public.chapters(book_id);

create index if not exists idx_story_versions_book_id
on public.story_versions(book_id);

create index if not exists idx_story_versions_parent
on public.story_versions(parent_version_id);

create index if not exists idx_story_graph_edges_book_id
on public.story_graph_edges(book_id);

create index if not exists idx_ai_jobs_user_id
on public.ai_jobs(user_id);

create index if not exists idx_story_analysis_book_id
on public.story_analysis(book_id);

create index if not exists idx_story_chunks_book_id
on public.story_chunks(book_id);

create index if not exists idx_story_entities_book_id
on public.story_entities(book_id);

create index if not exists idx_story_events_book_id
on public.story_events(book_id);

create index if not exists idx_prompt_contexts_book_id
on public.prompt_contexts(book_id);

create index if not exists idx_story_embeddings_book_id
on public.story_embeddings(book_id);

create index if not exists idx_story_embeddings_vector
on public.story_embeddings
using ivfflat (embedding vector_cosine_ops)
with (lists = 100);


create or replace function public.match_story_chunks(
  query_embedding vector(768),
  match_book_id uuid,
  match_count int default 8
)
returns table (
  chunk_id uuid,
  book_id uuid,
  content text,
  metadata jsonb,
  similarity float
)
language sql stable
as $$
  select
    sc.id as chunk_id,
    sc.book_id,
    sc.content,
    sc.metadata,
    1 - (se.embedding <=> query_embedding) as similarity
  from public.story_embeddings se
  join public.story_chunks sc on sc.id = se.chunk_id
  where se.book_id = match_book_id
  order by se.embedding <=> query_embedding
  limit match_count;
$$;
```
