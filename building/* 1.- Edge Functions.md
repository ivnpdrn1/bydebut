```bash
supabase secrets set GEMINI_API_KEY=TU_API_KEY_AQUI
```

```bash
supabase functions new semantic-search
```


```text
Project Settings
→ Edge Functions
→ Secrets
→ Add secret
```

Nombre:

```text
GEMINI_API_KEY
```

Valor:

```text
tu API key de Gemini
```

Después de guardar ese secreto, seguimos con esta primera función:

```text
create-embedding
```

Esta será la función que:

```text
1. Recibe entity_type y entity_id
2. Busca el contenido en Supabase
3. Construye el semantic_text
4. Envía ese semantic_text a Gemini
5. Guarda el embedding en la tabla embeddings
```

```ts
import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  try {
    const supabaseUrl = Deno.env.get("SUPABASE_URL");
    const serviceRoleKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY");
    const geminiApiKey = Deno.env.get("GEMINI_API_KEY");

    if (!supabaseUrl || !serviceRoleKey || !geminiApiKey) {
      throw new Error("Missing environment variables.");
    }

    const supabase = createClient(supabaseUrl, serviceRoleKey);

    const { entity_type, entity_id } = await req.json();

    if (!entity_type || !entity_id) {
      throw new Error("entity_type and entity_id are required.");
    }

    let semanticText = "";

    if (entity_type === "book") {
      const { data, error } = await supabase
        .from("books")
        .select("*")
        .eq("id", entity_id)
        .single();

      if (error) throw error;

      semanticText = `
Book Title: ${data.title || ""}
Subtitle: ${data.subtitle || ""}
Author: ${data.author_name || ""}
Genre: ${data.genre || ""}
Language: ${data.language || ""}
Description: ${data.description || ""}
`;
    }

    else if (entity_type === "chapter") {
      const { data, error } = await supabase
        .from("chapters")
        .select(`
          *,
          books (
            title,
            author_name,
            genre,
            language,
            description
          )
        `)
        .eq("id", entity_id)
        .single();

      if (error) throw error;

      semanticText = `
Book Title: ${data.books?.title || ""}
Author: ${data.books?.author_name || ""}
Genre: ${data.books?.genre || ""}
Language: ${data.books?.language || ""}
Book Description: ${data.books?.description || ""}

Chapter Number: ${data.chapter_number || ""}
Chapter Title: ${data.title || ""}
Chapter Summary: ${data.summary || ""}

Chapter Content:
${data.content || ""}
`;
    }

    else if (entity_type === "scene") {
      const { data, error } = await supabase
        .from("scenes")
        .select(`
          *,
          books (
            title,
            author_name,
            genre,
            language
          ),
          chapters (
            title,
            chapter_number,
            summary
          )
        `)
        .eq("id", entity_id)
        .single();

      if (error) throw error;

      semanticText = `
Book Title: ${data.books?.title || ""}
Author: ${data.books?.author_name || ""}
Genre: ${data.books?.genre || ""}
Language: ${data.books?.language || ""}

Chapter Number: ${data.chapters?.chapter_number || ""}
Chapter Title: ${data.chapters?.title || ""}
Chapter Summary: ${data.chapters?.summary || ""}

Scene Number: ${data.scene_number || ""}
Scene Title: ${data.title || ""}
Scene Summary: ${data.summary || ""}
Location: ${data.location || ""}
Time Period: ${data.time_period || ""}
Emotional Tone: ${data.emotional_tone || ""}

Scene Content:
${data.content || ""}
`;
    }

    else if (entity_type === "character") {
      const { data, error } = await supabase
        .from("characters")
        .select(`
          *,
          books (
            title,
            author_name,
            genre,
            language
          )
        `)
        .eq("id", entity_id)
        .single();

      if (error) throw error;

      semanticText = `
Book Title: ${data.books?.title || ""}
Author: ${data.books?.author_name || ""}
Genre: ${data.books?.genre || ""}
Language: ${data.books?.language || ""}

Character Name: ${data.name || ""}
Role in Story: ${data.role_in_story || ""}
Description: ${data.description || ""}
Motivation: ${data.motivation || ""}
Conflict: ${data.conflict || ""}
Arc Summary: ${data.arc_summary || ""}
`;
    }

    else if (entity_type === "story_version") {
      const { data, error } = await supabase
        .from("story_versions")
        .select(`
          *,
          books (
            title,
            author_name,
            genre,
            language
          ),
          chapters (
            title,
            chapter_number,
            summary
          ),
          scenes (
            title,
            summary,
            emotional_tone
          )
        `)
        .eq("id", entity_id)
        .single();

      if (error) throw error;

      semanticText = `
Book Title: ${data.books?.title || ""}
Author: ${data.books?.author_name || ""}
Genre: ${data.books?.genre || ""}
Language: ${data.books?.language || ""}

Chapter Number: ${data.chapters?.chapter_number || ""}
Chapter Title: ${data.chapters?.title || ""}
Chapter Summary: ${data.chapters?.summary || ""}

Scene Title: ${data.scenes?.title || ""}
Scene Summary: ${data.scenes?.summary || ""}
Scene Emotional Tone: ${data.scenes?.emotional_tone || ""}

Generated Version Title: ${data.title || ""}
Generation Mode: ${data.generation_mode || ""}
User Prompt:
${data.prompt || ""}

Generated Content:
${data.generated_content || ""}
`;
    }

    else {
      throw new Error("Invalid entity_type.");
    }

    const geminiResponse = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/text-embedding-004:embedContent?key=${geminiApiKey}`,
      {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: "models/text-embedding-004",
          content: {
            parts: [
              {
                text: semanticText,
              },
            ],
          },
        }),
      }
    );

    if (!geminiResponse.ok) {
      const errorText = await geminiResponse.text();
      throw new Error(`Gemini error: ${errorText}`);
    }

    const geminiData = await geminiResponse.json();
    const embedding = geminiData.embedding.values;

    const { data: savedEmbedding, error: saveError } = await supabase
      .from("embeddings")
      .insert({
        entity_type,
        entity_id,
        semantic_text: semanticText,
        embedding,
        embedding_model: "text-embedding-004",
        status: "mvp",
      })
      .select()
      .single();

    if (saveError) throw saveError;

    return new Response(
      JSON.stringify({
        success: true,
        entity_type,
        entity_id,
        embedding_id: savedEmbedding.id,
      }),
      {
        headers: {
          ...corsHeaders,
          "Content-Type": "application/json",
        },
      }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({
        success: false,
        error: error.message,
      }),
      {
        status: 400,
        headers: {
          ...corsHeaders,
          "Content-Type": "application/json",
        },
      }
    );
  }
});
```
