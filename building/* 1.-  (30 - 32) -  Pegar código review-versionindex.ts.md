````ts
import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  try {
    const supabaseUrl = Deno.env.get("SUPABASE_URL");
    const serviceRoleKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY");
    const geminiApiKey = Deno.env.get("GEMINI_API_KEY");

    if (!supabaseUrl || !serviceRoleKey || !geminiApiKey) {
      throw new Error("Missing environment variables.");
    }

    const supabase = createClient(supabaseUrl, serviceRoleKey);

    const { version_id, reviewer_type } = await req.json();

    if (!version_id) {
      throw new Error("version_id is required.");
    }

    const { data: version, error: versionError } = await supabase
      .from("story_versions")
      .select(`
        *,
        books (
          title,
          author_name,
          genre,
          language,
          description
        ),
        chapters (
          title,
          chapter_number,
          summary
        ),
        scenes (
          title,
          summary,
          emotional_tone
        )
      `)
      .eq("id", version_id)
      .single();

    if (versionError) throw versionError;

    const reviewPrompt = `
You are byDebut AI Reviewer.

Analyze the generated literary version below.

Reviewer Type:
${reviewer_type || "general"}

Book:
${version.books?.title || ""}

Author:
${version.books?.author_name || ""}

Genre:
${version.books?.genre || ""}

Language:
${version.books?.language || ""}

Book Description:
${version.books?.description || ""}

Chapter:
${version.chapters?.title || ""}

Chapter Summary:
${version.chapters?.summary || ""}

Scene:
${version.scenes?.title || ""}

Scene Summary:
${version.scenes?.summary || ""}

Emotional Tone:
${version.scenes?.emotional_tone || ""}

Generation Mode:
${version.generation_mode || ""}

User Prompt:
${version.prompt || ""}

Generated Content:
${version.generated_content || ""}

Return your response as valid JSON only.

Required JSON structure:
{
  "score": number from 1 to 10,
  "strengths": "text",
  "weaknesses": "text",
  "commercial_potential": "text",
  "literary_quality": "text",
  "cinematic_potential": "text",
  "ethical_notes": "text",
  "recommendation": "text"
}
`;

    const geminiResponse = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${geminiApiKey}`,
      {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          contents: [
            {
              parts: [
                {
                  text: reviewPrompt,
                },
              ],
            },
          ],
          generationConfig: {
            temperature: 0.35,
            topP: 0.9,
            maxOutputTokens: 2048,
          },
        }),
      }
    );

    if (!geminiResponse.ok) {
      const errorText = await geminiResponse.text();
      throw new Error(`Gemini error: ${errorText}`);
    }

    const geminiData = await geminiResponse.json();

    const rawText =
      geminiData?.candidates?.[0]?.content?.parts?.[0]?.text || "";

    if (!rawText) {
      throw new Error("Gemini returned empty review.");
    }

    let parsedReview;

    try {
      const cleaned = rawText
        .replace(/```json/g, "")
        .replace(/```/g, "")
        .trim();

      parsedReview = JSON.parse(cleaned);
    } catch (_e) {
      parsedReview = {
        score: null,
        strengths: "",
        weaknesses: "",
        commercial_potential: "",
        literary_quality: "",
        cinematic_potential: "",
        ethical_notes: "",
        recommendation: rawText,
      };
    }

    const { data: savedReview, error: saveError } = await supabase
      .from("ai_reviews")
      .insert({
        version_id,
        reviewer_type: reviewer_type || "general",
        score: parsedReview.score,
        strengths: parsedReview.strengths,
        weaknesses: parsedReview.weaknesses,
        commercial_potential: parsedReview.commercial_potential,
        literary_quality: parsedReview.literary_quality,
        cinematic_potential: parsedReview.cinematic_potential,
        ethical_notes: parsedReview.ethical_notes,
        recommendation: parsedReview.recommendation,
        raw_response: parsedReview,
      })
      .select()
      .single();

    if (saveError) throw saveError;

    await supabase.from("ai_logs").insert({
      user_id: version.creator_id,
      function_name: "review-version",
      model_name: "gemini-1.5-flash",
      request_payload: {
        version_id,
        reviewer_type: reviewer_type || "general",
      },
      response_payload: parsedReview,
    });

    return new Response(
      JSON.stringify({
        success: true,
        review: savedReview,
      }),
      {
        headers: {
          ...corsHeaders,
          "Content-Type": "application/json",
        },
      }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({
        success: false,
        error: error.message,
      }),
      {
        status: 400,
        headers: {
          ...corsHeaders,
          "Content-Type": "application/json",
        },
      }
    );
  }
});
````
