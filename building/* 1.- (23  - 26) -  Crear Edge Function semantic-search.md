## 24. Pegar código `semantic-search/index.ts`

Copia y pega esto:

```ts
import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  try {
    const supabaseUrl = Deno.env.get("SUPABASE_URL");
    const serviceRoleKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY");
    const geminiApiKey = Deno.env.get("GEMINI_API_KEY");

    if (!supabaseUrl || !serviceRoleKey || !geminiApiKey) {
      throw new Error("Missing environment variables.");
    }

    const supabase = createClient(supabaseUrl, serviceRoleKey);

    const body = await req.json();

    const query = body.query;
    const match_count = body.match_count || 10;
    const entity_type = body.entity_type || null;

    if (!query) {
      throw new Error("query is required.");
    }

    const geminiResponse = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/text-embedding-004:embedContent?key=${geminiApiKey}`,
      {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: "models/text-embedding-004",
          content: {
            parts: [
              {
                text: query,
              },
            ],
          },
        }),
      }
    );

    if (!geminiResponse.ok) {
      const errorText = await geminiResponse.text();
      throw new Error(`Gemini error: ${errorText}`);
    }

    const geminiData = await geminiResponse.json();
    const queryEmbedding = geminiData.embedding.values;

    const { data: matches, error: matchError } = await supabase.rpc(
      "match_embeddings",
      {
        query_embedding: queryEmbedding,
        match_count,
        filter_entity_type: entity_type,
      }
    );

    if (matchError) throw matchError;

    const enrichedResults = [];

    for (const item of matches || []) {
      let originalData = null;

      if (item.entity_type === "book") {
        const { data } = await supabase
          .from("books")
          .select("*")
          .eq("id", item.entity_id)
          .single();

        originalData = data;
      }

      if (item.entity_type === "chapter") {
        const { data } = await supabase
          .from("chapters")
          .select(`
            *,
            books (
              title,
              author_name,
              genre,
              language
            )
          `)
          .eq("id", item.entity_id)
          .single();

        originalData = data;
      }

      if (item.entity_type === "scene") {
        const { data } = await supabase
          .from("scenes")
          .select(`
            *,
            books (
              title,
              author_name,
              genre,
              language
            ),
            chapters (
              title,
              chapter_number
            )
          `)
          .eq("id", item.entity_id)
          .single();

        originalData = data;
      }

      if (item.entity_type === "character") {
        const { data } = await supabase
          .from("characters")
          .select(`
            *,
            books (
              title,
              author_name,
              genre,
              language
            )
          `)
          .eq("id", item.entity_id)
          .single();

        originalData = data;
      }

      if (item.entity_type === "story_version") {
        const { data } = await supabase
          .from("story_versions")
          .select(`
            *,
            books (
              title,
              author_name,
              genre,
              language
            ),
            chapters (
              title,
              chapter_number
            ),
            scenes (
              title,
              summary
            )
          `)
          .eq("id", item.entity_id)
          .single();

        originalData = data;
      }

      enrichedResults.push({
        match: item,
        original_data: originalData,
      });
    }

    return new Response(
      JSON.stringify({
        success: true,
        query,
        entity_type,
        match_count,
        results: enrichedResults,
      }),
      {
        headers: {
          ...corsHeaders,
          "Content-Type": "application/json",
        },
      }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({
        success: false,
        error: error.message,
      }),
      {
        status: 400,
        headers: {
          ...corsHeaders,
          "Content-Type": "application/json",
        },
      }
    );
  }
});
```

## 25. Probar `semantic-search`

En Supabase, usa el test de la función con este JSON:

```json
{
  "query": "stories about artificial intelligence, sacrifice, and power",
  "match_count": 5
}
```

También puedes probar filtrando por tipo:

```json
{
  "query": "a character sacrifices love for power",
  "match_count": 5,
  "entity_type": "chapter"
}
```

Importante: para que esta función devuelva resultados, antes debe existir al menos un embedding creado con:

```text
create-embedding
```
